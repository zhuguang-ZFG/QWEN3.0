Developer API

# 一句话生成绘图指令
import openai

client = openai.OpenAI(
  base_url="https://chat.donglicao.com/v1",
  api_key="your-key"
)

response = client.chat.completions.create(
  model="lima",
  messages=[{"role": "user",
    "content": "画一只猫"}]
)
