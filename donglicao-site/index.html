获取 API Key 
# 一句话生成绘图指令
import openai

client = openai.OpenAI(
  base_url="https://chat.donglicao.com/v1",
  api_key="your-key"
)

response = client.chat.completions.create(
  model="lima",
  messages=[{"role": "user",
    "content": "画一只猫"}]
)

print(response.choices[0].message.content)
curl -X POST "https://chat.donglicao.com/v1/chat/completions" \\
  -H "Content-Type: application/json" \\
  -H "Authorization: Bearer your-key" \\
  -d '{
    "model": "lima",
    "messages": [{"role": "user", "content": "画一只猫"}]
  }'
const response = await fetch('https://chat.donglicao.com/v1/chat/completions', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer your-key'
  },
  body: JSON.stringify({
    model: 'lima',
    messages: [{ role: 'user', content: '画一只猫' }]
  })
});

const data = await response.json();
console.log(data.choices[0].message.content);
package main

import (
  "bytes"
  "encoding/json"
  "fmt"
  "log"
  "net/http"
)

func main() {
  payload := map[string]any{
    "model": "lima",
    "messages": []map[string]string{
      {"role": "user", "content": "画一只猫"},
    },
  }
  body, _ := json.Marshal(payload)
  req, _ := http.NewRequest("POST",
    "https://chat.donglicao.com/v1/chat/completions",
    bytes.NewBuffer(body))
  req.Header.Set("Content-Type", "application/json")
  req.Header.Set("Authorization", "Bearer your-key")

  resp, err := http.DefaultClient.Do(req)
  if err != nil {
    log.Fatal(err)
  }
  defer resp.Body.Close()
  fmt.Println(resp.Status)
}
FAQ

常见问题

LiMa 支持哪些 AI 模型？

LiMa 量子星云已接入 170+ 后端，覆盖 GPT、Claude、Groq、NVIDIA、OpenRouter、DeepSeek、Cloudflare 等主流模型，并通过量子路由按请求类型、成本与健康状态坍缩至最优模型。

设备兼容哪些开发板？

AI 绘图机、写字机与 2D 数字人均基于 ESP32 系列硬件方案，支持 Wi-Fi / MQTT / WebSocket 接入。具体型号与固件可参考开发者文档。

数据是否安全？

LiMa 在传输层使用 TLS 加密，API 访问需校验 Key 与身份策略；设备指令经过策略验证与安全仿真，确保可执行且不会泄露隐私数据。

如何部署？

云端版本可直接使用 chat.donglicao.com，无需部署；本地或边缘部署提供 Docker 镜像与配置模板，按文档填写后端 Key 后即可启动服务。

支持私有化部署吗？

支持。企业客户可将 LiMa 路由核心、设备网关与数据库部署在自有服务器或私有云，满足合规与内网隔离要求，具体请联系商务团队。

如何计费？

按实际调用量计费，支持预付费套餐与按量后付。不同模型与能力单价不同，详情参考定价页面或控制台费率说明。

可以退款吗？

未消耗的预付费余额支持申请退款，购买后 7 天内且未使用可申请全额退；已消耗部分按实际使用结算，具体以购买协议为准。

支持多种设备协同吗？

支持。LiMa 的设备纠缠协同机制允许绘图机、写字机、数字人等多种设备在同一任务中联动，由路由中枢统一编排动作与状态。

API 有速率限制吗？

有。默认按账户级别设置 RPM 与 TPM 上限，免费体验额度较低；升级套餐或联系商务可提高限额。

固件如何升级？

设备支持 OTA 无线升级。LiMa 控制台推送新固件后，设备通过 MQTT 或 WebSocket 接收升级包并自动完成更新，也可使用 USB 线刷方式。

小程序支持哪些语言？

小程序 SDK 与示例目前支持中文与英文界面，语音交互默认支持中文普通话，后续将逐步扩展多语言 TTS 与 ASR。

如何获取技术支持？

可通过控制台工单、GitHub Issues 或发送邮件至 support@donglicao.com 获取技术支持，企业客户享有专属对接通道。
